\RestyleAlgo{ruled}
\SetKwComment{Comment}{/*}{*/}
\begin{algorithm}[h]
  \caption{An algorithm.}\label{alg:one}
  \KwData{$n\geq 0$}
  \KwResult{$y=x^n$}
  $y \gets 1$\;
  $X \gets x$\;
  $N \gets n$\;
  \While{$N\neq0$}{
    \eIf{$N$ is even}{
      $X\gets X\times X$\;
      $N \gets\frac{N}{2}$ \Comment*[r]{This is a comment}
    }{
      \If{$N$ is odd}{
        $y\gets y\times X$\;
        $N\gets N-1$\;
      }
    }
  }
\end{algorithm}
